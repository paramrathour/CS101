\begin{enumerate}
\item \textbf{Horner's method} an algorithm for polynomial evaluation\\
\begin{align}
\op{P}(x)=a_{0}&+a_{1}x+a_{2}x^{2}+a_{3}x^{3}+\cdots +a_{n}x^{n}\\
&=a_{0}+x\bigg(a_{1}+x\Big(a_{2}+x\big(a_{3}+\cdots +x(a_{n-1}+x\,a_{n})\cdots \big)\Big)\bigg)
\end{align}
This allows the evaluation of a polynomial of degree $n$ with only $n$ multiplications and $n$ additions.
This is optimal, since there are polynomials of degree $n$ that cannot be evaluated with fewer arithmetic operations

\textbf{Problem Statement:}\\
Write a function Horner() takes three inputs an array of coefficients $\op{P}$, degree of polynomial $n$ \& input $x$ and outputs the number $\op{P}(x)$ which is $\op{Horner}(\op{P},n,x)$\\
For $k$ testcases with coefficients $P_1[\ ], P_2[\ ],\ldots,P_n[\ ]$, degree $n_1,n_2,\ldots,n_k$ and variable $x_1,x_2,\ldots,x_n$ find $P_i(x_i)$ for each $i$ in $\{1,2,\ldots,n\}$\\
\begin{testcases}
    {$k$ (number of testcases)\\$n_i$ (degree of polynomial) $P_i[]$ ($n_i+1$ coefficients with $i$\textsuperscript{th} index coefficient of $i$\textsuperscript{th} power)  (each testcase on a newline)}
    {$\op{Horner}(\op{P},n,x)$ (each result on a newline)}
    {3\\0 5 1\\1 -3 2 2\\3 2 -1 -3 4}
    {5\\7\\80}
\end{testcases}
